### TASK 3: Database Persistence for Template Selection

**Goal**: Save user's template choice to database for both logged-in and guest users.

#### Step 3.1: Create API for Logged-In Users

**File**: `app/api/user/update-template/route.ts`

**Purpose**: Save template selection for authenticated users

**Code**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/server/db';
import { users } from '@/shared/schema';
import { eq } from 'drizzle-orm';
import { verifyToken } from '@/lib/auth/jwt';

export async function POST(request: NextRequest) {
  try {
    const token = request.headers.get('authorization')?.replace('Bearer ', '');
    if (!token) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const decoded = verifyToken(token);
    if (!decoded) {
      return NextResponse.json({ error: 'Invalid token' }, { status: 401 });
    }

    const { templateId } = await request.json();

    await db.update(users)
      .set({ selectedTemplate: templateId })
      .where(eq(users.id, decoded.userId));

    return NextResponse.json({ 
      success: true, 
      message: 'Template saved successfully' 
    });
  } catch (error) {
    console.error('Error saving template:', error);
    return NextResponse.json(
      { error: 'Failed to save template' },
      { status: 500 }
    );
  }
}
```

#### Step 3.2: Guest User Template Storage

**File**: `app/api/guest-cv/save-basic-info/route.ts`

**Purpose**: Save template + basic info for guest users

**Code**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/server/db';
import { guestCvs } from '@/shared/schema';
import { eq } from 'drizzle-orm';

export async function POST(request: NextRequest) {
  try {
    const { email, templateId, basicInfo } = await request.json();

    // Check if guest CV exists
    const existing = await db.select()
      .from(guestCvs)
      .where(eq(guestCvs.email, email))
      .limit(1);

    if (existing.length > 0) {
      // Update existing
      await db.update(guestCvs)
        .set({
          templateId,
          cvData: basicInfo,
          updatedAt: new Date()
        })
        .where(eq(guestCvs.email, email));
    } else {
      // Create new
      await db.insert(guestCvs).values({
        email,
        templateId,
        cvData: basicInfo,
        createdAt: new Date(),
        updatedAt: new Date()
      });
    }

    return NextResponse.json({ 
      success: true, 
      message: 'Data saved successfully' 
    });
  } catch (error) {
    console.error('Error saving guest CV:', error);
    return NextResponse.json(
      { error: 'Failed to save data' },
      { status: 500 }
    );
  }
}
```

```typescript
const handleSubmit = async (isAIGeneration: boolean) => {
  // 1. Validate form
  if (!validateForm()) return;

  // 2. Save to database
  if (token) {
    // Logged-in user
    await fetch('/api/user/update-template', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ templateId })
    });
  } else {
    // Guest user
    await fetch('/api/guest-cv/save-basic-info', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        email: formData.email,
        templateId,
        basicInfo: formData
      })
    });
  }

  // 3. Store in sessionStorage
  sessionStorage.setItem('cvBasicInfo', JSON.stringify(formData));
  sessionStorage.setItem('selectedTemplate', templateId);

  // 4. Redirect
  if (isAIGeneration) {
    router.push('/cv/preview?mode=ai');
  } else {
    router.push('/cv/edit');
  }
};
```

### TASK 5: Resume Capability for Guest Users

**Goal**: Allow returning guest users to resume their CV creation.

#### Step 5.1: Create Check API

**File**: `app/api/guest-cv/get-by-email/route.ts`

**Code**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/server/db';
import { guestCvs } from '@/shared/schema';
import { eq } from 'drizzle-orm';

export async function GET(request: NextRequest) {
  try {
    const email = request.nextUrl.searchParams.get('email');
    
    if (!email) {
      return NextResponse.json({ error: 'Email required' }, { status: 400 });
    }

    const guestCv = await db.select()
      .from(guestCvs)
      .where(eq(guestCvs.email, email))
      .limit(1);

    if (guestCv.length === 0) {
      return NextResponse.json({ found: false });
    }

    return NextResponse.json({
      found: true,
      data: guestCv[0]
    });
  } catch (error) {
    console.error('Error fetching guest CV:', error);
    return NextResponse.json(
      { error: 'Failed to fetch data' },
      { status: 500 }
    );
  }
}
```

#### Step 5.2: Implement Resume Logic

**In Basic Info Page**:
```typescript
// Check for existing CV when email is entered
const checkExistingCV = async (email: string) => {
  const response = await fetch(`/api/guest-cv/get-by-email?email=${email}`);
  const data = await response.json();

  if (data.found) {
    // Show confirmation dialog
    const resume = confirm(
      'We found a previous CV with this email. Would you like to resume?'
    );

    if (resume) {
      // Pre-fill form with existing data
      setFormData(data.data.cvData);
      setTemplateId(data.data.templateId);
    }
  }
};

// Trigger check on email blur
<input
  type="email"
  value={formData.email}
  onChange={(e) => setFormData({...formData, email: e.target.value})}
  onBlur={(e) => checkExistingCV(e.target.value)}
/>
```

2. Selects "Modern Professional" template

4. Redirected to `/cv/basic-info?template=modern-professional`
5. Form auto-filled from user profile
6. Updates any fields if needed
7. Clicks "Manual Continue"
8. Proceeds to CV builder with selected template

### Flow 3: Returning Guest User
1. User visits `/cv/basic-info`
2. Enters email address
3. System checks `guestCvs` table
4. Finds existing CV from 2 days ago
5. Shows confirmation: "Resume previous CV?"
6. User clicks "Yes"
7. Form pre-filled with previous data
8. Template restored from database
9. User continues editing

## 📊 DATABASE SCHEMA

### Templates Table
```sql
CREATE TABLE templates (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  isPremium BOOLEAN DEFAULT false,
  cssStyles JSONB,
  layout JSONB,
  previewImage VARCHAR(500),
  createdAt TIMESTAMP DEFAULT NOW()
);
```

### Users Table (Add Column)
```sql
ALTER TABLE users 
ADD COLUMN selectedTemplate VARCHAR(100);
```

### Guest CVs Table
```sql
CREATE TABLE guestCvs (
  id SERIAL PRIMARY KEY,
  email VARCHAR(255) NOT NULL,
  templateId VARCHAR(100),
  cvData JSONB,
  createdAt TIMESTAMP DEFAULT NOW(),
  updatedAt TIMESTAMP DEFAULT NOW()
);
```
